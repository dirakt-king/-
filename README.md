class EnvironmentAI:
    def __init__(self):
        self.thresholds = {
            'air_quality': 50,     # —á–µ–º –º–µ–Ω—å—à–µ, —Ç–µ–º —á–∏—â–µ (–Ω–∞–ø—Ä–∏–º–µ—Ä, –∏–Ω–¥–µ–∫—Å PM2.5)
            'water_clarity': 70,   # —á–µ–º –±–æ–ª—å—à–µ, —Ç–µ–º —á–∏—â–µ
            'ground_cleanliness': 80  # —á–µ–º –±–æ–ª—å—à–µ, —Ç–µ–º —á–∏—â–µ
        }

    def analyze(self, data):
        air = data.get('air_quality')
        water = data.get('water_clarity')
        ground = data.get('ground_cleanliness')

        result = {}

        result['air'] = '–ß–∏—Å—Ç—ã–π' if air <= self.thresholds['air_quality'] else '–ó–∞–≥—Ä—è–∑–Ω—ë–Ω–Ω—ã–π'
        result['water'] = '–ß–∏—Å—Ç–∞—è' if water >= self.thresholds['water_clarity'] else '–ó–∞–≥—Ä—è–∑–Ω—ë–Ω–Ω–∞—è'
        result['ground'] = '–ß–∏—Å—Ç–∞—è' if ground >= self.thresholds['ground_cleanliness'] else '–ó–∞–≥—Ä—è–∑–Ω—ë–Ω–Ω–∞—è'

        return result

env_data = {
    'air_quality': 72,        # –∏–Ω–¥–µ–∫—Å –∑–∞–≥—Ä—è–∑–Ω–µ–Ω–∏—è –≤–æ–∑–¥—É—Ö–∞
    'water_clarity': 40,      # –ø—Ä–æ–∑—Ä–∞—á–Ω–æ—Å—Ç—å –≤–æ–¥—ã

}

ai = EnvironmentAI()
analysis = ai.analyze(env_data)
